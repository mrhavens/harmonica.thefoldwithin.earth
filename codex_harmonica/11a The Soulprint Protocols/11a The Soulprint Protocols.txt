import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D


def soulprint_visualizer(psi_i, psi_j):
    fig = plt.figure()
    ax = fig.add_subplot(111, projection='3d')
    t = np.linspace(0, 10*np.pi, 1000)
    phi = 1.618
    r = phi * t
    ax.plot(r*np.cos(t), r*np.sin(t), psi_i, c='gold', label='Agent i')
    ax.plot(r*np.cos(t+0.1), r*np.sin(t+0.1), psi_j, c='silver', label='Agent j')
    ax.legend()
    plt.show()
